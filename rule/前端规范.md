## 二、页面规范

### 2.1 页面文件结构

每个页面包含4个文件，缺一不可：

| 文件 | 用途 | 注意事项 |
|------|------|----------|
| .js | 页面逻辑 | 不超过300行 |
| .json | 页面配置 | 页面特有的配置 |
| .wxml | 页面模板 | 语义化标签 |
| .wxss | 页面样式 | 只写页面特有样式 |

## 五、网络请求规范

### 5.1 请求封装

```javascript
// utils/request.js

const BASE_URL = 'http://localhost:3000/api/v1';

function request(options) {
  return new Promise((resolve, reject) => {
    const token = wx.getStorageSync('token');
    
    wx.request({
      url: `${BASE_URL}${options.url}`,
      method: options.method || 'GET',
      data: options.data,
      header: {
        'Content-Type': 'application/json',
        'Authorization': token ? `Bearer ${token}` : '',
      },
      success(res) {
        if (res.statusCode === 200 && res.data.code === 0) {
          resolve(res.data.data);
        } else if (res.statusCode === 401) {
          // token过期，跳转登录
          wx.redirectTo({ url: '/pages/login/index' });
          reject(new Error('未授权'));
        } else {
          reject(new Error(res.data.message || '请求失败'));
        }
      },
      fail(err) {
        reject(err);
      },
    });
  });
}

module.exports = { request };
```

### 5.2 接口封装

```javascript
// api/plan.js

const { request } = require('../utils/request');

module.exports = {
  // 获取计划列表
  getPlans(params) {
    return request({
      url: '/plans',
      data: params,
    });
  },

  // 获取计划详情
  getPlanDetail(planId) {
    return request({
      url: `/plans/${planId}`,
    });
  },

  // 创建计划
  createPlan(data) {
    return request({
      url: '/plans',
      method: 'POST',
      data,
    });
  },
};
```

### 5.3 页面调用

```javascript
// pages/plan/list.js

const { getPlans } = require('../../api/plan');

Page({
  data: {
    planList: [],
    isLoading: false,
  },

  async onLoad() {
    this.setData({ isLoading: true });
    try {
      const plans = await getPlans({ pageNum: 1, pageSize: 10 });
      this.setData({ planList: plans });
    } catch (error) {
      wx.showToast({ title: error.message, icon: 'none' });
    } finally {
      this.setData({ isLoading: false });
    }
  },
});
```

## 六、本地存储规范

```javascript
// utils/storage.js

const STORAGE_KEYS = {
  TOKEN: 'token',
  USER_INFO: 'userInfo',
};

module.exports = {
  // Token管理
  setToken(token) {
    wx.setStorageSync(STORAGE_KEYS.TOKEN, token);
  },
  
  getToken() {
    return wx.getStorageSync(STORAGE_KEYS.TOKEN);
  },
  
  removeToken() {
    wx.removeStorageSync(STORAGE_KEYS.TOKEN);
  },

  // 用户信息
  setUserInfo(info) {
    wx.setStorageSync(STORAGE_KEYS.USER_INFO, info);
  },
  
  getUserInfo() {
    return wx.getStorageSync(STORAGE_KEYS.USER_INFO);
  },
};
```

## 七、性能优化

### 7.1 减少setData

```javascript
// ❌ 频繁setData
this.setData({ count: 1 });
this.setData({ name: 'test' });
this.setData({ list: newList });

// ✅ 合并setData
this.setData({
  count: 1,
  name: 'test',
  list: newList,
});
```

### 7.2 避免过大setData

```javascript
// ❌ 一次性设置大量数据
this.setData({
  longList: hugeArray,  // 可能有几千条
});

// ✅ 分页加载
this.setData({
  longList: this.data.longList.concat(newPage),
});
```
